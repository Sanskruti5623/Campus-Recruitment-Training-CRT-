#include<iostream>
using namespace std;

class stack {
private:
    int size;
    int *arr;
    int top;

public:
    stack (int s) {
        int size = s;
        arr = new int [size];
        top = -1;
    }
    bool isEmpty() {
      return top ==-1;
    }
    bool isFull () {
        return top ==size-1;
    }
    void push (int value) {
       if (isFull()) {
           cout << "Stack is Full" <<endl;
           return;
       }
       arr[++top] = value;
       cout << "Element pushed =" << arr[top] <<endl;
    }
    void pop (){
        if(isEmpty()){
            cout<<"Stack is Empty"<<endl;
            return;
        }
        cout << "Element popped =" << arr[top] <<endl;
        top--;
    }
    void peek() {
        if(isEmpty()) {
            cout<<"Stack is Empty"<<endl;
        }
        cout<<"Top element = "<<arr[top]<<endl;
    }

    void display () {
        if (isEmpty ()) {
            cout << "Stack is empty" <<endl;
            return;
        }
        for (int i = top; i >= 0; i--) {
            cout <<"[ "<<arr[i]<<" ]" <<endl;
        }
    }
};

int main() {
   stack st(5);
   int ch;
   cout <<"Stack has been created with capacity 5" <<endl;
   while (true) {
        cout << "\n1. Push" << endl;
        cout << "2. Pop" << endl;
        cout << "3. Display Stack" << endl;
        cout << "4. Peek" << endl;
        cout << "5. Exit" << endl;
        cout << "Enter your choice: ";
        cin >> ch;

        if (ch == 1) {
            int val;
            cout << "Enter value to be pushed: ";
            cin >> val;
            st.push(val);
        }
        else if (ch == 2) {
            st.pop();
        }
        else if (ch == 3) {
            st.display();
        }
        else if (ch == 4) {
            st.peek();
        }
        else if (ch == 5) {
            cout << "Exiting program..." << endl;
            break;
        }
        else {
            cout << "Invalid choice!" << endl;
        }
    }
}
