2. Big omega: it is a complexity that is going to be atleast more than the best.
3. Big Theta: it s the complexity within bounds of the worsth and the best cases.

#include<iostream>
using namespace std;
int linearSearch(int arr[],int size,int target){
for(int i=0; i<size;i++){
    if (arr[i]==target){
        return i;
    }
}
return -1;
}
int main(){
    int count=0;
    int arr[]={3,4,6,7,2,3,2,4,2,9};
    int size=sizeof(arr)/sizeof(arr[0]);
    int target =2;
    
    
    int result = linearSearch(arr,size,target);
